```python
import io
import json

import pandas as pd

from config import ANOMALY_THRESHOLD_SIGMA

TOOL_DEFINITION = {
    "name": "anomaly_detect",
    "description": "Detect statistical anomalies in time series data using rolling Z-score",
    "input_schema": {
        "type": "object",
        "properties": {
            "csv_data": {"type": "string", "description": "CSV with date and metric columns"},
            "metric_column": {"type": "string"},
            "date_column": {"type": "string"},
        },
        "required": ["csv_data", "metric_column", "date_column"],
    },
}


def detect_anomalies(
    csv_data: str,
    metric_column: str,
    date_column: str,
    sigma: float = ANOMALY_THRESHOLD_SIGMA,
    min_points: int = 5,
) -> dict:
    df = pd.read_csv(io.StringIO(csv_data))
    if metric_column not in df.columns or date_column not in df.columns:
        return {"anomalies": [], "error": f"Column not found"}
    if len(df) < min_points:
        return {"anomalies": []}

    df = df.sort_values(date_column).reset_index(drop=True)
    series = df[metric_column].astype(float)

    window = min(7, len(df) - 1)
    rolling_mean = series.rolling(window=window, min_periods=3, center=True).mean()
    rolling_std = series.rolling(window=window, min_periods=3, center=True).std()

    anomalies = []
    for i, (val, mean, std) in enumerate(zip(series, rolling_mean, rolling_std)):
        if pd.isna(mean) or pd.isna(std) or std == 0:
            continue
        zscore = (val - mean) / std
        if abs(zscore) > sigma:
            if abs(zscore) > 4:
                severity = "severe"
            elif abs(zscore) > 3:
                severity = "moderate"
            else:
                severity = "mild"
            anomalies.append({
                "date": str(df[date_column].iloc[i]),
                "value": float(val),
                "zscore": round(float(zscore), 2),
                "severity": severity,
                "direction": "up" if zscore > 0 else "down",
            })

    return {"anomalies": anomalies}


def handle_anomaly_detect(csv_data: str, metric_column: str, date_column: str) -> str:
    result = detect_anomalies(csv_data, metric_column, date_column)
    return json.dumps(result, ensure_ascii=False)
```

```python
import io
import json
from datetime import datetime, timedelta

import numpy as np
import pandas as pd
from scipy import stats

TOOL_DEFINITION = {
    "name": "trend_forecast",
    "description": "Forecast future metric values using linear regression on recent data",
    "input_schema": {
        "type": "object",
        "properties": {
            "csv_data": {"type": "string"},
            "metric_column": {"type": "string"},
            "date_column": {"type": "string"},
            "forecast_days": {"type": "integer", "default": 7},
        },
        "required": ["csv_data", "metric_column", "date_column"],
    },
}

_TREND_SLOPE_THRESHOLD = 0.005


def forecast_trend(
    csv_data: str,
    metric_column: str,
    date_column: str,
    forecast_days: int = 7,
    fit_window: int = 14,
) -> dict:
    df = pd.read_csv(io.StringIO(csv_data))
    df = df.sort_values(date_column).tail(fit_window).reset_index(drop=True)

    x = np.arange(len(df))
    y = df[metric_column].astype(float).values

    slope, intercept, r_value, _, _ = stats.linregress(x, y)
    predicted_in_sample = slope * x + intercept
    residuals = y - predicted_in_sample
    std_residual = np.std(residuals)
    ci_multiplier = 1.645  # 90% confidence

    last_date_str = str(df[date_column].iloc[-1])
    try:
        last_date = datetime.strptime(last_date_str, "%Y%m%d")
        date_fmt = "%Y%m%d"
    except ValueError:
        last_date = datetime.strptime(last_date_str, "%Y-%m-%d")
        date_fmt = "%Y-%m-%d"

    forecast = []
    for i in range(1, forecast_days + 1):
        x_future = len(df) - 1 + i
        pred = slope * x_future + intercept
        margin = ci_multiplier * std_residual
        future_date = last_date + timedelta(days=i)
        forecast.append({
            "date": future_date.strftime(date_fmt),
            "predicted": round(float(pred)),
            "lower": round(float(pred - margin)),
            "upper": round(float(pred + margin)),
        })

    normalized_slope = slope / (np.mean(y) if np.mean(y) != 0 else 1)
    if normalized_slope > _TREND_SLOPE_THRESHOLD:
        trend_direction = "up"
    elif normalized_slope < -_TREND_SLOPE_THRESHOLD:
        trend_direction = "down"
    else:
        trend_direction = "flat"

    return {
        "forecast": forecast,
        "trend_direction": trend_direction,
        "r_squared": round(float(r_value ** 2), 3),
    }


def handle_trend_forecast(
    csv_data: str,
    metric_column: str,
    date_column: str,
    forecast_days: int = 7,
) -> str:
    result = forecast_trend(csv_data, metric_column, date_column, forecast_days)
    return json.dumps(result, ensure_ascii=False)
```

```python
import json
from datetime import datetime
from pathlib import Path

from config import EVENTS_FILE, EVENT_CORRELATE_WINDOW_DAYS

TOOL_DEFINITION = {
    "name": "event_correlate",
    "description": "Correlate anomaly dates with business events (version releases, operations)",
    "input_schema": {
        "type": "object",
        "properties": {
            "anomaly_dates": {
                "type": "array",
                "items": {"type": "string"},
                "description": "List of anomaly dates in YYYYMMDD format",
            },
        },
        "required": ["anomaly_dates"],
    },
}


def _parse_date(date_str: str) -> datetime:
    for fmt in ("%Y%m%d", "%Y-%m-%d"):
        try:
            return datetime.strptime(date_str, fmt)
        except ValueError:
            continue
    raise ValueError(f"Cannot parse date: {date_str}")


def correlate_events(
    anomaly_dates: list[str],
    events: list[dict],
    window_days: int = EVENT_CORRELATE_WINDOW_DAYS,
) -> dict:
    if not anomaly_dates:
        return {"correlations": []}

    correlations = []
    for anomaly_date_str in anomaly_dates:
        anomaly_dt = _parse_date(anomaly_date_str)
        matches = []
        for event in events:
            event_dt = _parse_date(event["date"])
            diff = abs((anomaly_dt - event_dt).days)
            if diff <= window_days:
                matches.append((diff, event))
        matches.sort(key=lambda x: x[0])
        for diff, event in matches:
            correlations.append({
                "anomaly_date": anomaly_date_str,
                "matched_event": event,
                "days_diff": diff,
            })

    return {"correlations": correlations}


def handle_event_correlate(anomaly_dates: list[str]) -> str:
    try:
        events = json.loads(Path(EVENTS_FILE).read_text(encoding="utf-8"))
    except (FileNotFoundError, json.JSONDecodeError):
        events = []
    result = correlate_events(anomaly_dates, events)
    return json.dumps(result, ensure_ascii=False)
```

```python
import json
import re
import subprocess
from datetime import datetime, timedelta

KYUUBI_REGION = "chnbj"
KYUUBI_WORKSPACE = "11329"

COMPARE_DAYS = {"wow": 7, "mom": 28, "yoy": 365}

TOOL_DEFINITION = {
    "name": "period_compare",
    "description": "Compare current period with a previous period (wow/mom/yoy) by re-running the same SQL with shifted dates",
    "input_schema": {
        "type": "object",
        "properties": {
            "sql": {"type": "string", "description": "Original nl-sql-generated query"},
            "compare_type": {"type": "string", "enum": ["wow", "mom", "yoy"]},
            "metric_column": {"type": "string", "description": "Name of the metric column"},
        },
        "required": ["sql", "compare_type", "metric_column"],
    },
}


def shift_dates_in_sql(sql: str, days_back: int) -> str:
    def replace_date(m: re.Match) -> str:
        try:
            d = datetime.strptime(m.group(1), "%Y%m%d") - timedelta(days=days_back)
            return f"'{d.strftime('%Y%m%d')}'"
        except ValueError:
            return m.group(0)
    return re.sub(r"'(\d{8})'", replace_date, sql)


def run_kyuubi_query(sql: str) -> dict:
    result = subprocess.run(
        ["kyuubi", "sql", "query", sql, "--region", KYUUBI_REGION, "--workspace", KYUUBI_WORKSPACE],
        capture_output=True,
        text=True,
        timeout=300,
    )
    if result.returncode != 0:
        return {"error": result.stderr.strip() or f"kyuubi exit code {result.returncode}"}
    try:
        return json.loads(result.stdout)
    except json.JSONDecodeError:
        return {"raw_output": result.stdout.strip()}


def handle_period_compare(sql: str, compare_type: str, metric_column: str) -> str:
    days = COMPARE_DAYS.get(compare_type, 7)
    comparison_sql = shift_dates_in_sql(sql, days)
    current_result = run_kyuubi_query(sql)
    previous_result = run_kyuubi_query(comparison_sql)
    return json.dumps({
        "compare_type": compare_type,
        "days_back": days,
        "metric_column": metric_column,
        "current_result": current_result,
        "previous_result": previous_result,
    }, ensure_ascii=False, default=str)
```

```python
import json
import subprocess

KYUUBI_REGION = "chnbj"
KYUUBI_WORKSPACE = "11329"

TOOL_DEFINITION = {
    "name": "auto_drilldown",
    "description": "Drill down by one or more dimensions to find top contributors to a metric",
    "input_schema": {
        "type": "object",
        "properties": {
            "sql": {"type": "string", "description": "Original nl-sql-generated query"},
            "dimensions": {
                "type": "array",
                "items": {"type": "string"},
                "description": "Dimension column names to drilldown by",
            },
            "metric_column": {"type": "string", "description": "Metric column to aggregate", "default": "dau"},
        },
        "required": ["sql", "dimensions"],
    },
}


def generate_drilldown_sql(original_sql: str, dimension: str, metric_column: str = "dau") -> str:
    inner = original_sql.rstrip(";").rstrip()
    return (
        f"SELECT {dimension}, SUM({metric_column}) AS {metric_column} "
        f"FROM ({inner}) t "
        f"GROUP BY {dimension} "
        f"ORDER BY {metric_column} DESC "
        f"LIMIT 20"
    )


def run_kyuubi_query(sql: str) -> dict:
    result = subprocess.run(
        ["kyuubi", "sql", "query", sql, "--region", KYUUBI_REGION, "--workspace", KYUUBI_WORKSPACE],
        capture_output=True,
        text=True,
        timeout=300,
    )
    if result.returncode != 0:
        return {"error": result.stderr.strip() or f"kyuubi exit code {result.returncode}"}
    try:
        return json.loads(result.stdout)
    except json.JSONDecodeError:
        return {"raw_output": result.stdout.strip()}


def handle_auto_drilldown(sql: str, dimensions: list[str], metric_column: str = "dau") -> str:
    drilldowns = []
    for dimension in dimensions:
        drilldown_sql = generate_drilldown_sql(sql, dimension, metric_column)
        query_result = run_kyuubi_query(drilldown_sql)
        drilldowns.append({
            "dimension": dimension,
            "result": query_result,
        })
    return json.dumps({"drilldowns": drilldowns}, ensure_ascii=False, default=str)
```

```python
import asyncio
import json
from dataclasses import dataclass, field

from openai import AsyncOpenAI

from config import MODEL_MAIN, API_BASE_URL, API_KEY, MAX_ANALYSIS_ROUNDS
from tools.anomaly_detect import handle_anomaly_detect
from tools.trend_forecast import handle_trend_forecast
from tools.event_correlate import handle_event_correlate
from tools.period_compare import handle_period_compare
from tools.auto_drilldown import handle_auto_drilldown

client = AsyncOpenAI(base_url=API_BASE_URL, api_key=API_KEY)

ANALYSIS_PLAN_PROMPT = """\
You are an analysis planner for a data analysis agent. Given the initial query results and the user's goal, decide which supplementary analyses should be run.

Available actions:
1. period_compare — run if YoY/WoW/MoM comparison would add value (params: compare_type: wow|mom|yoy, metric_column)
2. auto_drilldown — run if there is notable variation to attribute to dimensions (params: dimensions: list of column names, metric_column)
3. anomaly_detect — run if the data is a time series with ≥5 data points (params: metric_column, date_column)
4. trend_forecast — run if future trajectory is relevant to the user's goal (params: metric_column, date_column, forecast_days)
5. event_correlate — run ONLY if anomalies have already been detected (params: {})

Rules:
- Only include actions that materially improve the analysis
- Do not repeat actions already executed
- Output must be valid JSON: {"steps": [...]}
- If no further analysis is needed, output: {"steps": []}

Output format example:
{"steps": [
  {"action": "anomaly_detect", "params": {"metric_column": "dau", "date_column": "date"}},
  {"action": "period_compare", "params": {"compare_type": "wow", "metric_column": "dau"}}
]}
"""


@dataclass
class AnalysisPackage:
    initial_data: str
    sql_used: str
    period_comparison: dict = field(default_factory=dict)
    drilldown_results: list = field(default_factory=list)
    anomalies: list = field(default_factory=list)
    forecast: dict = field(default_factory=dict)
    event_correlations: list = field(default_factory=list)
    executed_actions: set = field(default_factory=set)

    def summary(self) -> str:
        parts = [f"Initial data (first 600 chars):\n{self.initial_data[:600]}"]
        if self.period_comparison:
            parts.append(f"Period comparison:\n{json.dumps(self.period_comparison)[:400]}")
        if self.anomalies:
            parts.append(f"Anomalies detected:\n{json.dumps(self.anomalies)[:400]}")
        if self.forecast:
            parts.append(f"Forecast:\n{json.dumps(self.forecast)[:300]}")
        if self.drilldown_results:
            parts.append(f"Drilldown results:\n{json.dumps(self.drilldown_results)[:400]}")
        return "\n\n".join(parts)

    def to_context_string(self) -> str:
        return self.summary()


def execute_analysis_step(step: dict, sql_used: str, package: AnalysisPackage) -> None:
    action = step["action"]
    params = step.get("params", {})

    if action == "period_compare":
        raw = handle_period_compare(
            sql=sql_used,
            compare_type=params.get("compare_type", "wow"),
            metric_column=params.get("metric_column", "dau"),
        )
        package.period_comparison = json.loads(raw)

    elif action == "auto_drilldown":
        raw = handle_auto_drilldown(
            sql=sql_used,
            dimensions=params.get("dimensions", []),
            metric_column=params.get("metric_column", "dau"),
        )
        package.drilldown_results.append(json.loads(raw))

    elif action == "anomaly_detect":
        raw = handle_anomaly_detect(
            csv_data=package.initial_data,
            metric_column=params.get("metric_column", "dau"),
            date_column=params.get("date_column", "date"),
        )
        data = json.loads(raw)
        package.anomalies = data.get("anomalies", [])

    elif action == "trend_forecast":
        raw = handle_trend_forecast(
            csv_data=package.initial_data,
            metric_column=params.get("metric_column", "dau"),
            date_column=params.get("date_column", "date"),
            forecast_days=params.get("forecast_days", 7),
        )
        package.forecast = json.loads(raw)

    elif action == "event_correlate":
        anomaly_dates = [a["date"] for a in package.anomalies]
        raw = handle_event_correlate(anomaly_dates=anomaly_dates)
        package.event_correlations = json.loads(raw).get("correlations", [])

    package.executed_actions.add(action)


async def generate_analysis_plan(data_summary: str, analysis_brief: str, already_done: set) -> list[dict]:
    done_str = ", ".join(already_done) if already_done else "none"
    response = await client.chat.completions.create(
        model=MODEL_MAIN,
        max_tokens=512,
        messages=[
            {"role": "system", "content": ANALYSIS_PLAN_PROMPT},
            {"role": "user", "content": (
                f"Data summary:\n{data_summary}\n\n"
                f"User goal: {analysis_brief}\n\n"
                f"Already executed: {done_str}"
            )},
        ],
        response_format={"type": "json_object"},
    )
    try:
        result = json.loads(response.choices[0].message.content or "{}")
        return result.get("steps", [])
    except (json.JSONDecodeError, AttributeError):
        return []


async def run_analysis_engine(
    initial_data: str,
    sql_used: str,
    analysis_brief: str,
) -> AnalysisPackage:
    package = AnalysisPackage(initial_data=initial_data, sql_used=sql_used)

    for _ in range(MAX_ANALYSIS_ROUNDS):
        plan = await generate_analysis_plan(package.summary(), analysis_brief, package.executed_actions)
        new_steps = [s for s in plan if s["action"] not in package.executed_actions]
        if not new_steps:
            break
        for step in new_steps:
            execute_analysis_step(step, sql_used, package)

    return package
```

```python
async def run_debate(
    analysis_package,  # AnalysisPackage from analysis_engine
    analysis_brief: str,
) -> dict:
    from experts.analysis_engine import AnalysisPackage
    data_context = (
        f"## 分析目标\n\n{analysis_brief}\n\n"
        f"## 完整分析数据\n\n{analysis_package.to_context_string()}"
    )

    # Round 1: initial opinions (parallel)

        run_expert(name, prompt, data_context)
        for name, prompt in EXPERT_CONFIGS

        challenges = []

            others = [o for o in latest_opinions if o["role"] != name]
            ctx = data_context + f"\n\n## 其他专家的观点\n\n{_format_opinions(others)}"
            challenge = await run_expert(name, CHALLENGE_PROMPT, ctx)
            challenges.append(challenge)

        responses = []
        for name, _ in EXPERT_CONFIGS:
            my_challenges = [c for c in challenges if c["role"] != name]
            my_opinion = next(o for o in latest_opinions if o["role"] == name)
            ctx = (
                data_context
                + f"\n\n## 你的观点\n\n{my_opinion['analysis']}"
                + f"\n\n## 对你的质疑\n\n{_format_opinions(my_challenges)}"
            )
            response = await run_expert(name, RESPONSE_PROMPT, ctx)
            responses.append(response)

        all_rounds_context += (
            f"\n\n## Round {debate_round + 2} 质疑\n\n{_format_opinions(challenges)}"
            f"\n\n## Round {debate_round + 2} 回应\n\n{_format_opinions(responses)}"
        )

        # Arbiter verdict
        is_last_round = debate_round == MAX_DEBATE_ROUNDS - 1
        verdict = await run_arbiter(
            data_context + "\n\n" + all_rounds_context,
            force=is_last_round,
        )
        if verdict.get("converged", True):
            return verdict

        latest_opinions = responses

    return verdict
```

```python
def _build_report(state: dict, package, verdict: dict) -> str:
    from experts.analysis_engine import AnalysisPackage
    parts = ["# 深度数据分析报告\n"]

    # Summary
    rec = verdict.get("final_recommendation", "")
    if rec:
        parts.append(f"## 摘要\n\n{rec}\n")

    # Consensus
    consensus = verdict.get("consensus_conclusions", [])
    if consensus:
        parts.append("## 专家共识\n\n" + "\n".join(f"- {c}" for c in consensus) + "\n")

    # Period comparison
    if package.period_comparison:
        parts.append(f"## 同比/环比数据\n\n```json\n{json.dumps(package.period_comparison, ensure_ascii=False, indent=2)}\n```\n")

    # Anomalies
    if package.anomalies:
        rows = ["| 日期 | 值 | Z-score | 严重程度 | 方向 |", "|------|-----|---------|----------|------|"]
        for a in package.anomalies:
            rows.append(f"| {a['date']} | {a['value']:,} | {a['zscore']} | {a['severity']} | {a['direction']} |")
        parts.append("## 异常检测\n\n" + "\n".join(rows) + "\n")

    # Event correlations
    if package.event_correlations:
        parts.append("## 事件关联\n")
        for c in package.event_correlations:
            evt = c["matched_event"]
            parts.append(f"- 异常日期 {c['anomaly_date']} ← {c['days_diff']}天前: {evt['description']} ({evt['type']})")
        parts.append("")

    # Forecast
    if package.forecast.get("forecast"):
        parts.append(f"## 趋势预测\n\n趋势方向: **{package.forecast.get('trend_direction', 'unknown')}**\n")
        rows = ["| 日期 | 预测值 | 低区间 | 高区间 |", "|------|--------|--------|--------|"]
        for p in package.forecast["forecast"]:
            rows.append(f"| {p['date']} | {p['predicted']:,} | {p['lower']:,} | {p['upper']:,} |")
        parts.append("\n".join(rows) + "\n")

    # Disagreements
    disagreements = verdict.get("disagreements", [])
    if disagreements:
        parts.append("## 专家分歧（供参考）\n")
        for d in disagreements:
            parts.append(f"**{d['topic']}**")
            for role, pos in d.get("positions", {}).items():
                parts.append(f"  - {role}: {pos}")
        parts.append("")

    # SQL appendix
    parts.append(f"## 附录：使用的 SQL\n\n```sql\n{state.get('sql_used', '')}\n```\n")

    return "\n".join(parts)
```
